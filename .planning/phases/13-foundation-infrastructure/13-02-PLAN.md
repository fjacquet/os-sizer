```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest'
import { setActivePinia, createPinia } from 'pinia'
import { useCalculationStore } from '../calculationStore'
import { useInputStore } from '../inputStore'
import type { ClusterSizing } from '@/engine/types'

// Mock the engine to return controlled sizing results
vi.mock('@/engine/index', () => ({
  calcCluster: vi.fn(),
  recommend: vi.fn().mockReturnValue([]),
  validateInputs: vi.fn().mockReturnValue([]),
}))

import { calcCluster } from '@/engine/index'
const mockCalcCluster = vi.mocked(calcCluster)

function makeSizing(totals: { vcpu: number; ramGB: number; storageGB: number }): { sizing: ClusterSizing } {
  return {
    sizing: {
      masterNodes: { count: 3, vcpu: 8, ramGB: 32, storageGB: 120 },
      workerNodes: null,
      infraNodes: null,
      odfNodes: null,
      rhacmWorkers: null,
      virtWorkerNodes: null,
      gpuNodes: null,
      virtStorageGB: 0,
      rhoaiOverhead: null,
      totals,
    },
  }
}

describe('calculationStore.aggregateTotals', () => {
  beforeEach(() => {
    setActivePinia(createPinia())
  })

  it('returns zero totals when no clusters exist', () => {
    mockCalcCluster.mockReturnValue(makeSizing({ vcpu: 0, ramGB: 0, storageGB: 0 }))
    const calc = useCalculationStore()
    // Even with default single cluster, if sizing is zero
    expect(calc.aggregateTotals).toEqual({ vcpu: 0, ramGB: 0, storageGB: 0 })
  })

  it('returns totals for a single cluster', () => {
    mockCalcCluster.mockReturnValue(makeSizing({ vcpu: 104, ramGB: 416, storageGB: 1360 }))
    const calc = useCalculationStore()
    expect(calc.aggregateTotals).toEqual({ vcpu: 104, ramGB: 416, storageGB: 1360 })
  })

  it('sums totals across two clusters', () => {
    mockCalcCluster
      .mockReturnValueOnce(makeSizing({ vcpu: 100, ramGB: 400, storageGB: 1000 }))
      .mockReturnValueOnce(makeSizing({ vcpu: 50, ramGB: 200, storageGB: 500 }))

    const input = useInputStore()
    input.addCluster()  // adds a second cluster
    const calc = useCalculationStore()
    expect(calc.aggregateTotals).toEqual({ vcpu: 150, ramGB: 600, storageGB: 1500 })
  })

  it('sums totals across three clusters', () => {
    mockCalcCluster
      .mockReturnValueOnce(makeSizing({ vcpu: 100, ramGB: 200, storageGB: 300 }))
      .mockReturnValueOnce(makeSizing({ vcpu: 10, ramGB: 20, storageGB: 30 }))
      .mockReturnValueOnce(makeSizing({ vcpu: 1, ramGB: 2, storageGB: 3 }))

    const input = useInputStore()
    input.addCluster()
    input.addCluster()
    const calc = useCalculationStore()
    expect(calc.aggregateTotals).toEqual({ vcpu: 111, ramGB: 222, storageGB: 333 })
  })
})
```
