```typescript
// ── makeSizing helper (local to this test file, below existing imports/helpers) ──
function makeSizing(overrides?: {
  workerNodes?: NodeSpec | null
  infraNodes?: NodeSpec | null
}): ClusterSizing {
  return {
    masterNodes: { count: 3, vcpu: 4, ramGB: 16, storageGB: 100 },
    workerNodes: overrides?.workerNodes !== undefined
      ? overrides.workerNodes
      : { count: 3, vcpu: 4, ramGB: 8, storageGB: 100 },
    infraNodes: overrides?.infraNodes !== undefined
      ? overrides.infraNodes
      : null,
    odfNodes: null,
    rhacmWorkers: null,
    virtWorkerNodes: null,
    gpuNodes: null,
    virtStorageGB: 0,
    totals: { vcpu: 0, ramGB: 0, storageGB: 0 },
  }
}

describe('calcRHOAI — worker floor enforcement (RHOAI-02)', () => {
  it('lifts worker vcpu and ramGB below floor up to RHOAI_WORKER_MIN_VCPU / RHOAI_WORKER_MIN_RAM_GB', () => {
    const sizing = makeSizing({ workerNodes: { count: 3, vcpu: 4, ramGB: 8, storageGB: 100 } })
    calcRHOAI(sizing, false)
    expect(sizing.workerNodes?.vcpu).toBe(RHOAI_WORKER_MIN_VCPU)
    expect(sizing.workerNodes?.ramGB).toBe(RHOAI_WORKER_MIN_RAM_GB)
  })

  it('does not change worker vcpu/ramGB already at the minimum (at-minimum case)', () => {
    const sizing = makeSizing({ workerNodes: { count: 3, vcpu: 8, ramGB: 32, storageGB: 100 } })
    calcRHOAI(sizing, false)
    expect(sizing.workerNodes?.vcpu).toBe(8)
    expect(sizing.workerNodes?.ramGB).toBe(32)
  })

  it('does not lower worker vcpu/ramGB already above the minimum', () => {
    const sizing = makeSizing({ workerNodes: { count: 3, vcpu: 32, ramGB: 64, storageGB: 100 } })
    calcRHOAI(sizing, false)
    expect(sizing.workerNodes?.vcpu).toBe(32)
    expect(sizing.workerNodes?.ramGB).toBe(64)
  })

  it('is a no-op and does not throw when workerNodes is null (SNO / compact-3node)', () => {
    const sizing = makeSizing({ workerNodes: null })
    expect(() => calcRHOAI(sizing, false)).not.toThrow()
    expect(sizing.workerNodes).toBeNull()
  })
})

describe('calcRHOAI — infra node overhead addend (RHOAI-03)', () => {
  it('adds RHOAI_INFRA_OVERHEAD_VCPU and RHOAI_INFRA_OVERHEAD_RAM_GB to infra nodes when infraNodesEnabled=true', () => {
    const infraNodes = { count: 3, vcpu: 4, ramGB: 24, storageGB: 100 }
    const sizing = makeSizing({ infraNodes })
    calcRHOAI(sizing, true)
    expect(sizing.infraNodes?.vcpu).toBe(4 + RHOAI_INFRA_OVERHEAD_VCPU)
    expect(sizing.infraNodes?.ramGB).toBe(24 + RHOAI_INFRA_OVERHEAD_RAM_GB)
    expect(sizing.infraNodes?.count).toBe(3)           // count must not change
    expect(sizing.infraNodes?.storageGB).toBe(100)     // storageGB must not change
  })

  it('does not modify infra nodes when infraNodesEnabled=false', () => {
    const infraNodes = { count: 3, vcpu: 4, ramGB: 24, storageGB: 100 }
    const sizing = makeSizing({ infraNodes })
    calcRHOAI(sizing, false)
    expect(sizing.infraNodes?.vcpu).toBe(4)
    expect(sizing.infraNodes?.ramGB).toBe(24)
  })
})

describe('RHOAI constants — RHOAI 3.x cluster minimum verification (success criterion 3)', () => {
  it('RHOAI_WORKER_MIN_VCPU is 8 — matches RHOAI 3.x documented per-worker minimum', () => {
    expect(RHOAI_WORKER_MIN_VCPU).toBe(8)
  })

  it('RHOAI_WORKER_MIN_RAM_GB is 32 — matches RHOAI 3.x documented per-worker minimum', () => {
    expect(RHOAI_WORKER_MIN_RAM_GB).toBe(32)
  })

  it('2 workers at minimum produce 16 vCPU and 64 GB cluster total — RHOAI 3.x minimum cluster', () => {
    expect(2 * RHOAI_WORKER_MIN_VCPU).toBe(16)
    expect(2 * RHOAI_WORKER_MIN_RAM_GB).toBe(64)
  })
})
```
