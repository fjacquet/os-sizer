**validation.ts** -- Create input validation module:

```typescript
import type { ClusterConfig, ValidationWarning } from './types'
import { WORKER_MIN } from './constants'

/**
 * Validate ClusterConfig inputs and return warnings/errors.
 * Never throws — always returns ValidationWarning[].
 * All messageKey values are i18n keys (no English strings).
 */
export function validateInputs(config: ClusterConfig): ValidationWarning[] {
  const warnings: ValidationWarning[] = []

  // Negative pod count
  if (config.workload.totalPods < 0) {
    warnings.push({
      code: 'NEGATIVE_PODS',
      severity: 'error',
      messageKey: 'validation.negativePods',
    })
  }

  // Worker node below minimum RAM
  if (config.workload.nodeRamGB < WORKER_MIN.ramGB) {
    warnings.push({
      code: 'WORKER_RAM_BELOW_MIN',
      severity: 'warning',
      messageKey: 'validation.workerRamBelowMin',
    })
  }

  // Worker node below minimum CPU
  if (config.workload.nodeVcpu < WORKER_MIN.vcpu) {
    warnings.push({
      code: 'WORKER_CPU_BELOW_MIN',
      severity: 'warning',
      messageKey: 'validation.workerCpuBelowMin',
    })
  }

  // Zero pods with non-zero CPU/RAM (likely user error)
  if (config.workload.totalPods === 0 && (config.workload.podCpuMillicores > 0 || config.workload.podMemMiB > 0)) {
    warnings.push({
      code: 'ZERO_PODS_WITH_RESOURCES',
      severity: 'warning',
      messageKey: 'validation.zeroPodsWithResources',
    })
  }

  // ODF enabled on non-HA topologies
  if (config.addOns.odfEnabled && ['sno', 'microshift', 'managed-cloud'].includes(config.topology)) {
    warnings.push({
      code: 'ODF_INCOMPATIBLE_TOPOLOGY',
      severity: 'error',
      messageKey: 'validation.odfIncompatibleTopology',
    })
  }

  // RHACM with 0 managed clusters
  if (config.addOns.rhacmEnabled && config.addOns.rhacmManagedClusters <= 0) {
    warnings.push({
      code: 'RHACM_NO_CLUSTERS',
      severity: 'warning',
      messageKey: 'validation.rhacmNoClusters',
    })
  }

  return warnings
}
```

All messageKey values are i18n keys. No English string literals in warnings.

**validation.test.ts** -- Replace stubs with real tests:

```typescript
import { describe, it, expect } from 'vitest'
import { readdirSync, readFileSync } from 'node:fs'
import { join } from 'node:path'
import { validateInputs } from './validation'
import { createDefaultClusterConfig } from './defaults'

describe('engine zero-Vue-import constraint (ENG-09)', () => {
  it('no engine .ts file imports from vue, pinia, or vue-i18n', () => {
    const engineDir = join(__dirname)
    const tsFiles = readdirSync(engineDir)
      .filter(f => f.endsWith('.ts') && !f.endsWith('.test.ts'))

    for (const file of tsFiles) {
      const content = readFileSync(join(engineDir, file), 'utf-8')
      expect(content, `${file} imports vue`).not.toMatch(/from\s+['"]vue['"]/)
      expect(content, `${file} imports pinia`).not.toMatch(/from\s+['"]pinia['"]/)
      expect(content, `${file} imports vue-i18n`).not.toMatch(/from\s+['"]vue-i18n['"]/)
    }
  })
})

describe('validateInputs', () => {
  it('returns empty array for valid default config', () => {
    const config = createDefaultClusterConfig(0)
    const result = validateInputs(config)
    expect(result).toEqual([])
  })

  it('returns error for negative pod count', () => {
    const config = createDefaultClusterConfig(0)
    config.workload.totalPods = -1
    const result = validateInputs(config)
    expect(result.some(w => w.code === 'NEGATIVE_PODS' && w.severity === 'error')).toBe(true)
  })

  it('returns warning for worker RAM below minimum', () => {
    const config = createDefaultClusterConfig(0)
    config.workload.nodeRamGB = 4  // below WORKER_MIN.ramGB (8)
    const result = validateInputs(config)
    expect(result.some(w => w.code === 'WORKER_RAM_BELOW_MIN')).toBe(true)
  })

  it('returns error for ODF on SNO topology', () => {
    const config = createDefaultClusterConfig(0)
    config.topology = 'sno'
    config.addOns.odfEnabled = true
    const result = validateInputs(config)
    expect(result.some(w => w.code === 'ODF_INCOMPATIBLE_TOPOLOGY')).toBe(true)
  })

  it('all messageKeys are i18n keys without spaces', () => {
    const config = createDefaultClusterConfig(0)
    config.workload.totalPods = -1
    config.workload.nodeRamGB = 1
    const result = validateInputs(config)
    for (const w of result) {
      expect(w.messageKey).toMatch(/^\S+$/)
    }
  })
})
```
