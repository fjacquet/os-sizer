```typescript
import type { NodeSpec } from './types'
import { ODF_MIN_CPU_PER_NODE, ODF_MIN_RAM_PER_NODE_GB, ODF_MIN_NODES, ODF_CPU_PER_OSD, ODF_RAM_PER_OSD_GB } from './constants'
import { infraNodeSizing } from './formulas'

/**
 * ODF storage node sizing.
 * Base: 3 nodes x 16 vCPU / 64 GB. Per extra OSD: +2 CPU / +5 GB per node.
 * Source: hardware-sizing.md section 5.1
 * @param extraOsdCount - additional OSDs beyond base (0 = minimum)
 */
export function calcODF(extraOsdCount: number): NodeSpec {
  return {
    count: ODF_MIN_NODES,
    vcpu: ODF_MIN_CPU_PER_NODE + (extraOsdCount * ODF_CPU_PER_OSD),
    ramGB: ODF_MIN_RAM_PER_NODE_GB + (extraOsdCount * ODF_RAM_PER_OSD_GB),
    storageGB: 0,  // ODF storage is on raw block devices, not counted in OS disk
  }
}

/**
 * Infrastructure node sizing based on worker count.
 * Always 3 infra nodes for HA. Size per node from infra sizing table.
 * Source: hardware-sizing.md section 4
 */
export function calcInfraNodes(workerCount: number): NodeSpec {
  const spec = infraNodeSizing(workerCount)
  return {
    count: 3,
    vcpu: spec.vcpu,
    ramGB: spec.ramGB,
    storageGB: 100,
  }
}

/**
 * RHACM hub cluster worker sizing.
 * Two-tier: < 100 managed clusters -> 3 x 8vCPU/32GB; >= 100 -> 3 x 16vCPU/64GB.
 * Source: hardware-sizing.md section 5.2
 */
export function calcRHACM(managedClusterCount: number): NodeSpec {
  if (managedClusterCount < 100) {
    return { count: 3, vcpu: 8, ramGB: 32, storageGB: 100 }
  }
  return { count: 3, vcpu: 16, ramGB: 64, storageGB: 100 }
}
```

All functions return `NodeSpec` directly. The topology calculators (Plan 02-03) call these when `addOns.odfEnabled`, `addOns.infraNodesEnabled`, or `addOns.rhacmEnabled` are true.
